LISTA DE EXERCICIOS DE LIKE Buscas incompletas

--exe 1
 SELECT * FROM fornecedores
WHERE SUBSTRING(telefone,2,2) IN('11','21');

-- exe 2
SELECT descricao, CONCAT(cidade,'/',estado) as 'cidade/estado'
FROM fornecedores WHERE estado IN('RJ', 'MG', 'SP');

-- exe 3
SELEC * FROM produtos
WHERE principio LIKE '____' AND principio NOT LIKE'% %';

-- exe 4
SELEC * FROM produtos
WHERE principio LIKE 'A%' AND principio NOT LIKE'% % % %';

-- exe 5
SELEC * FROM clientes
WHERE principio LIKE '____11-%' AND SUBSTRING(cidade,1,1)= 'M';

-- exe 6
SELECT nome, CONCAT(SUBSTRING(datacadasto,6,2),'/', SUBSTRING(datacadasto,1,4)) as 'mes/ano'
FROM clientes
WHERE NOT(cidade='NOVA FRIBURGO' OR cidade='CORDEIRO');

-- exe 7
SELECT DISTINT CONCAT(cidade,'-', bairro) as 'cidade/bairro'
FROM  clientes
WHERE 	cidade!=''AND cidade is not null AND bairro<>'' AND bairro is not null;

-- exe 8
SELECT*, (pcusto*0,05)as lucro FROM produtos
WHERE (pcusto*0,05)>30;

-- exe 9
SELEC * FROM produtos
WHERE estoque !='' AND estoque is not null AND apresentacao<>'' AND apresentacao is not null;
